Turtle Crossing Game

Crossing game I designed using the Python Turtle library. I used the Turtle library because I thought it would be effective to grasp the features of this library with this method. I added levels so that as the level increases so does the speed of the boxes. The logic is simple. If the boxes touch turtle it's game over.

The code I wrote for this project:

            
#main.py

from turtle import Screen, Turtle, onkey
import time
from cars import Cars
from scoreboard import Scoreboard
screen=Screen()
from turtle_player import Player
screen.setup(600,600)
screen.tracer(0)
player=Player()
screen.listen()
cars=Cars()
scoreboard=Scoreboard()
screen.onkey(player.move_up,"Up")
continue_game=True
while continue_game:
    time.sleep(0.1)
    screen.update()
    cars.create_car()
    cars.moving_cars()
    for car in cars.all_cars:
        if car.distance(player)<21:
            continue_game=False
            scoreboard.game_over()
    if player.finish_line():
        player.return_start()
        cars.level_up()
        scoreboard.increase_level()
screen.exitonclick()

#scoreboard.py

from turtle import Turtle
FONT=["Courier",15,"bold"]
class Scoreboard(Turtle):
    def __init__(self):
        super().__init__()
        self.level=1
        self.hideturtle()
        self.penup()
        self.goto(-285,270)
        self.update_scoreboard()
    def update_scoreboard(self):
        self.clear()
        self.write(f"Level: {self.level}",align="left",font=FONT)
    def increase_level(self):
        self.level+=1
        self.update_scoreboard()
    def game_over(self):
        self.clear()
        self.goto(0,0)
        self.write(f"GAME OVER",align="center",font=FONT)
        self.goto(0,-20)
        self.write(f"Your Highest Level: {self.level}",align="center",font=FONT)

#cars.py

from turtle import Turtle
import random
colors=["red","orange","yellow","green","blue","purple","black","grey","brown"]
class Cars(Turtle):
    def __init__(self):
        self.all_cars=[]
        self.car_speed=5
    def create_car(self):
        chance=random.randint(1,5)
        if chance==1:
            new_car=Turtle("square")
            new_car.penup()
            new_car.shapesize(1,2)
            new_car.color(random.choice(colors))
            random_y=random.randint(-250,250)
            new_car.goto(300,random_y)
            self.all_cars.append(new_car)
    def moving_cars(self):
        for car in self.all_cars:
            car.backward(self.car_speed)
    def level_up(self):
        self.car_speed+=3

#turtle_player.py

from turtle import Turtle
class Player(Turtle):
    def __init__(self):
        super().__init__()
        self.shape("turtle")
        self.penup()
        self.return_start()
        self.setheading(90)
    def move_up(self):
        self.forward(10)
    def return_start(self):
        self.goto(0,-280)
    def finish_line(self):
        if self.ycor()>280:
            return True
        else:
            return False
